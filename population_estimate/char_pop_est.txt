### Model 

model {

  # mean abundance
  for (i in 1:l) {
    lambda[i] ~ dgamma(lp[1, 1], lp[1, 2])
  } # i
  
  # overdispersion
  r ~ dgamma(rl[1, 1], rl[1, 2])
  
  # detection probability
  for (i in 1:p) {
    phi[i] ~ dbeta(pp[1, 1], pp[1, 2])
  } # i

  # abundance
  for (i in 1:n) {
    nu_l[i] ~ dgamma(r, r / lambda[lambda_nu[i]])
    nu[i] ~ dpois(nu_l[i])
  } # i
  
  # detection
  for (i in 1:length(y)) {
    y[i] ~ dbinom(phi[phi_y[i]], nu[nu_y[i]])
  } # i


}
